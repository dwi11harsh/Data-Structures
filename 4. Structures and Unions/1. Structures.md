- Structures are used for grouping of elements of different data types.
```c
#include <stdio.h>

struct {
    int i;
    char ch;
} x, y, z;

int main() {
    x.i = 12;
    x.ch = 'x';
    
    y.i = 14;
    y.ch = 'y';
    
    z.i = 17;
    z.ch = 'z';
    
    printf("x: i = %d, c = %c\n", x.i, x.ch);
    printf("y: i = %d, c = %c\n", y.i, y.ch);
    printf("z: i = %d, c = %c\n", z.i, z.ch);

    return 0;
}
```
